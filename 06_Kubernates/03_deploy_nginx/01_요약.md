# 1) nginx 실행 deployment 만들기(수정)
# 2) 실행하기:
kubectl apply -f nginx.yml

# 3) 등록 확인하기:
kubectl get all

# 4) my-nginx 상태 상세 확인하기:
kubectl describe deployment my-nginx

# 5) my-nginx 문제 있을때 실행(에러포함) 로그 보기:
kubectl logs -l app=my-nginx

# 6) 웹 브라우저 접속하기
minikube service my-nginx-service

# 7) deploy 삭제
kubectl delete deployment my-nginx
kubectl delete service my-nginx-service
