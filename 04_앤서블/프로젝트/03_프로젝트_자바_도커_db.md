# db01
---
- name: Setup DB Server
  hosts: db
  become: yes

  tasks:

    - name: Install Java
      apt:
        name: openjdk-17-jdk
        state: present
        update_cache: yes

    - name: Install Docker
      apt:
        name: docker.io
        state: present

    - name: Start Docker
      systemd:
        name: docker
        state: started
        enabled: yes

    - name: Run Oracle XE container
      shell: |
        docker run -d \
        --name oracle-xe \
        -p 1521:1521 \
        -e ORACLE_PASSWORD=1234 \
        gvenzl/oracle-xe:18
      args:
        executable: /bin/bash
