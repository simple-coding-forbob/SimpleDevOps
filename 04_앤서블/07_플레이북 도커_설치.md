# 도커 설치 : docker.yml
---
# Playbook 제목
- name: Install and Start Docker on Local EC2
  # 실행 대상 (inventory의 그룹명)
  hosts: local
  # sudo 권한 필요
  become: yes

  tasks:
    # 1️⃣ Docker 패키지 설치
    - name: Install docker package
      # Ubuntu 패키지 관리 모듈
      apt:
        # 설치할 패키지 이름
        name: docker.io
        # Docker가 설치된 상태로 유지
        state: present
        # apt update 실행
        update_cache: yes


    # 2️⃣ Docker 서비스 시작
    - name: Start Docker service
      # 서비스 제어 모듈
      systemd:
        # 서비스 이름
        name: docker
        # 실행 상태 유지
        state: started
        # 재부팅 시 자동 시작
        enabled: yes
