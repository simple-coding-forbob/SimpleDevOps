# playbook 작성: jdk.yml
---
# 이 Playbook의 이름 (설명용)
- name: Install JDK on Local EC2
  # 실행 대상 (inventory 파일의 그룹명)
  hosts: local
  # sudo 권한으로 실행 (root 권한 필요)
  become: yes
  # 실제 작업 목록
  tasks:
    # 작업 1 : JDK 설치
    - name: Install OpenJDK 17
      # apt 모듈 사용 (Ubuntu 패키지 설치 모듈)
      apt:
        # 설치할 패키지 이름
        name: openjdk-17-jdk
        # 상태 정의
        # present = 설치하라는 의미
        # absent  = 삭제하라는 의미
        state: present
        # 패키지 목록 최신화 (apt update 역할)
        update_cache: yes
