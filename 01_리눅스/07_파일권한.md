# 파일 권한 바꾸기
# 1) 문자 방식 2) 숫자 방식
# 문자 방식으로 실습(쉬움)
# 사용법: chmod 대상+권한 파일명
## 대상: 자신(u), 그룹(g), 타인(o), 모두(a)
## 권한: 읽기(r), 쓰기(w), 실행(x)

# 예제) readme.txt 파일의 그룹(g)에 쓰기(w) 권한 주기
chmod g+w readme.txt
ls -al

# 예제) readme.txt 파일에 그룹(g)에 쓰기 권한(w) 뺒기
chmod g-w readme.txt
ls -al

# 예제) readme.txt 파일에 모든사람에(a) 쓰기 권한 주기
chmod a+w readme.txt
ls -al
